```js
import React, { useState } from "react";
import "./App.css";

const App = () => {
  const [count, setCount] = useState(0);
  return (
    <div className="App">
      {count}
      <button onClick={() => setCount(count + 1)}>Increment</button>
    </div>
  );
};

export default App;
}
```

Or you can an increment function separately instead of an inline

```js
import React, { useState } from "react"
import "./App.css"

const App = () => {
  const [count, setCount] = useState(0)
  //or you can create an increment function here instead of inline
  const increment = () => {
    setCount(count + 1)
  }
  return (
    <div className="App">
      {count}
      <button onClick={increment}>Increment</button>
    </div>
  )
}

export default App
```

Adding Decrement

```js
import React, { useState } from "react"
import "./App.css"

const App = () => {
  const [count, setCount] = useState(0)
  //or you can create an increment function here instead of inline
  const increment = () => {
    setCount(count + 1)
  }
  const decrement = () => {
    setCount(count - 1)
  }
  return (
    <div className="App">
      {count}
      <button onClick={increment}>Increment</button>
      <button onClick={decrement}>Derement</button>
    </div>
  )
}

export default App
```
